#include <stdio.h>
#include <stdlib.h>

typedef struct TreeNode
{
	int data, id;
	struct TreeNode* left, * right;
}TreeNode;

TreeNode* makeNode(int size,int id, TreeNode* left, TreeNode* right) // 노드의 주소값이 반환됨
{
	TreeNode* node;
	node = (TreeNode*)malloc(sizeof(TreeNode));
	node->data = size;
	node->id = id;
	node->left = left;
	node->right = right;
	return node;
}

void folderSize(TreeNode* root)
{
	printf("%d ", root->data);
	if (root->left)
		printf("%d ", root->left->data);
	if (root->right)
		printf("%d ", root->right->data);
}

void preorder(TreeNode* root)
{
	if (root)
	{
		printf(" %d", root->data);
		preorder(root->left);
		preorder(root->right);
	}
}

void inorder(TreeNode* root)
{
	if (root)
	{
		
		inorder(root->left);
		printf(" %d", root->data);
		inorder(root->right);
	}
}

void postorder(TreeNode* root)
{
	if (root)
	{
		
		postorder(root->left);
		postorder(root->right);
		printf(" %d", root->data);
	}
}

TreeNode* findnodeid(TreeNode* root, int id)
{
	TreeNode* p;
	
	if (root)
	{
		if (id == root->id)
			return root;

		p = findnodeid(root->left,id);
		if (p != NULL)
			return p;

		p = findnodeid(root->right, id);
		if (p != NULL)
			return p;

	}
	return NULL;
}

void print(TreeNode*root, int order)
{
	if (order == 1)
		preorder(root);
	else if (order == 2)
		inorder(root);
	else if (order == 3)
		postorder(root);
	printf("\n");
}

{
	int n,id;
	TreeNode* tree,*p;

	TreeNode* F4 = makeNode(70,4, NULL, NULL);
	TreeNode* F5 = makeNode(90,5, NULL, NULL);
	TreeNode* F2 = makeNode(30,2, F4, F5);

	TreeNode* F7 = makeNode(130, 7,NULL, NULL);
	TreeNode* F8 = makeNode(80, 8,NULL, NULL);
	TreeNode* F6 = makeNode(120, 6,F7, F8);
	TreeNode* F3 = makeNode(50, 3,NULL, F6);

	TreeNode* F1 = makeNode(20, 1,F2, F3);

	tree = F1;

	scanf("%d %d", &n,&id);
	p = findnodeid(tree, id);

	if (p == NULL)
		printf("-1\n");
	else
		print(p, n);

	return 0;

}
